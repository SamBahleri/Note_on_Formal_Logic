\section*{Fitch-style Rules for First-Order Logic}

\subsection*{1.1 Universal Elimination ($\forall$E)}
\noindent
\begin{minipage}{0.48\textwidth}
\fbox{\begin{minipage}{\dimexpr\textwidth-2\fboxsep-2\fboxrule\relax}
\begin{nd}
\hypo{1}{$\forall x\, Px$}
\have{2}{$Pa$} \by{$\forall$E}{1}
\end{nd}
\end{minipage}}
\end{minipage}
\hfill
\begin{minipage}{0.48\textwidth}
\fbox{\begin{minipage}{\dimexpr\textwidth-2\fboxsep-2\fboxrule\relax}
\begin{nd}
\hypo{1}{$\forall x(P(x) \land Q(x))$}
\have{2}{$P(c) \land Q(c)$} \by{$\forall$E}{1}
\end{nd}
\end{minipage}}
\end{minipage}

\subsection*{1.2 Existential Introduction ($\exists$I)}

\begin{nd}

\end{minipage}

\begin{minipage}{0.48\textwidth}

\end{nd}

\end{minipage}

\subsection*{1.3 Universal Introduction ($\forall$I)}
\noindent
\begin{minipage}{0.48\textwidth}
\fbox{\begin{minipage}{\dimexpr\textwidth-2\fboxsep-2\fboxrule\relax}
\begin{nd}
\subproof{
\hypo{1}{$Pa$}
\have{2}{$\vdots$}
\have{3}{$Qa$}
}
\have{4}{$\forall x\, Qx$} \by{$\forall$I}{1-3}
\end{nd}
\end{minipage}}
\end{minipage}

\subsection*{1.4 Existential Elimination ($\exists$E)}
\noindent
\begin{minipage}{0.48\textwidth}
\fbox{\begin{minipage}{\dimexpr\textwidth-2\fboxsep-2\fboxrule\relax}
\begin{nd}
\hypo{1}{$\exists x\, Px$}
\subproof{
\hypo{2}{$Pa$}
\have{3}{$\vdots$}
\have{4}{$Q$}
}
\have{5}{$Q$} \by{$\exists$E}{1,2-4}
\end{nd}
\end{minipage}}
\end{minipage}
